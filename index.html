رحلة إلى حمام السخنة
""" # Save HTML file html_path = os.path.join(base_path, "index.html") with open(html_path, "w", encoding="utf-8") as f: f.write(html_content) # Create zip file zip_path = "/mnt/data/website_ready_mobile_carousel.zip" with ZipFile(zip_path, "w") as zipf: for foldername, subfolders, filenames in os.walk(base_path): for filename in filenames: file_path = os.path.join(foldername, filename) arcname = os.path.relpath(file_path, base_path) zipf.write(file_path, arcname=arcname) zip_path
